PRAC 01 Simple Calculator – Addition, Subtraction, Multiplication, Division
Aim

Write a Kotlin program that demonstrates the simple calculator with Addition, Subtraction, Multiplication and Division operations.

Kotlin 
fun main() {
    Add(10, 30)
    Sub(40, 30)
    Mul(50, 10)
    Div(100, 10)
}

fun Add(a: Int, b: Int) {
    println("Addition of $a and $b = ${a + b}")
}

fun Sub(a: Int, b: Int) {
    println("Subtraction of $a and $b = ${a - b}")
}

fun Mul(a: Int, b: Int) {
    println("Multiplication of $a and $b = ${a * b}")
}

fun Div(a: Int, b: Int) {
    if (b != 0)
        println("Division of $a and $b = ${a / b}")
    else
        println("Cannot divide by zero")
}
PRAC 02 OOP Concepts – Classes, Objects, Constructors & Methods
Aim

Write a Kotlin program demonstrating Object-Oriented Programming concepts by creating classes, objects, constructors, and methods to store and display student details.

Kotlin 
class Student(var name: String, var rollNo: Int, var marks: Int) {
    fun displayDetails() {
        println("Student Name: $name")
        println("Roll Number: $rollNo")
        println("Marks: $marks")
    }
}

fun main() {
    val student1 = Student("sandeep", 101, 85)
    val student2 = Student("rahul", 102, 90)

    println("Student 1 Details:")
    student1.displayDetails()

    println()

    println("Student 2 Details:")
    student2.displayDetails()
}

Kotlin · Android 
package com.cys.myapplication

import android.os.Bundle
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat
import android.widget.*

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)

        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }

        val num1 = findViewById<EditText>(R.id.num1)
        val num2 = findViewById<EditText>(R.id.num2)
        val result = findViewById<TextView>(R.id.result)
        val add = findViewById<Button>(R.id.btnAdd)
        val sub = findViewById<Button>(R.id.btnSub)
        val mul = findViewById<Button>(R.id.btnMul)
        val div = findViewById<Button>(R.id.btnDiv)

        add.setOnClickListener {
            val a = num1.text.toString().toInt()
            val b = num2.text.toString().toInt()
            result.text = "Result: ${a + b}"
        }

        sub.setOnClickListener {
            val a = num1.text.toString().toInt()
            val b = num2.text.toString().toInt()
            result.text = "Result: ${a - b}"
        }

        mul.setOnClickListener {
            val a = num1.text.toString().toInt()
            val b = num2.text.toString().toInt()
            result.text = "Result: ${a * b}"
        }

        div.setOnClickListener {
            val a = num1.text.toString().toInt()
            val b = num2.text.toString().toInt()
            if (b != 0) result.text = "Result: ${a / b}"
            else result.text = "Error: Cannot divide by zero"
        }
    }
}

package com.example.firstapp

import android.os.Bundle
import android.util.Log
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        Log.e("FirstApp", "This is onCreate() method")
    }

    override fun onStart() {
        super.onStart()
        Log.e("FirstApp", "This is onStart() method")
    }

    override fun onResume() {
        super.onResume()
        Log.e("FirstApp", "This is onResume() method")
    }

    override fun onPause() {
        super.onPause()
        Log.e("FirstApp", "This is onPause() method")
    }

    override fun onStop() {
        super.onStop()
        Log.e("FirstApp", "This is onStop() method")
    }

    override fun onDestroy() {
        super.onDestroy()
        Log.e("FirstApp", "This is onDestroy() method")
    }

    override fun onRestart() {
        super.onRestart()
        Log.e("FirstApp", "This is onRestart() method")
    }
}
